#include <stdio.h>
#include <stdlib.h>

void linearSearch(char *str, char item, int index) {
    if (str[index] == '\0') {
        printf("The item '%c' is not found in the string.\n", item);
        return;
    }

    if (str[index] == item) {
        printf("The item '%c' is found at index %d in the string.\n", item, index);
        return;
    }

    linearSearch(str, item, index + 1);
}

int main() {

    printf("Enter a string: ");
    gets(str);
    
    printf("Enter a character to search: ");

    return 0;
}

#include <stdio.h>	
#include <stdlib.h>

void arrayinput(int *arr, int size) {
    int i;
    printf("\n");
    for(i = 0; i < size; i++) {
        printf("Enter the array element arr[%d] :", i);
        scanf("%d", &arr[i]);
    }
}

void arrayouput(int *arr, int size) {
    int i;
    printf("\n");
    printf("OUTPUT : ");
    printf("\n");
    for(i = 0; i < size; i++) {
        printf("\t%d", arr[i]);
    }
}

void arraybinary(int *arr, int size, int item) {
    int min, max, mid;
    min = 0;
    max = size - 1;

    if (item < arr[0] || item > arr[size - 1]) {
        printf("Item %d is not found", item);	 
        return;
    } else {
        while(min <= max) {
            mid = (min + max) / 2;

            if(item == arr[mid]) {
                printf("The item %d is found at the index arr[%d] in binary search.", item, mid);
                break;  	
            }  
            else if(item < arr[mid]) {
                max = mid - 1;
            }	
            else if(item > arr[mid]) {
                min = mid + 1;
            }
        }
    }
}

int main() {
    int size, item;
    int *arr;

    printf("Enter the array size: ");
    scanf("%d", &size);

    arr = (int *) malloc(sizeof(int) * size);

    arrayinput(arr, size);
    arrayouput(arr, size);

    printf("\nEnter the item to search: ");
    scanf("%d", &item);

    arraybinary(arr, size, item);

    free(arr);
    return 0;
}

Question 6: Binary Search with character throught recursion

#include <stdio.h>
#include <stdlib.h>

void sortString(char *str, int size) {
    char temp;
    for (int i = 0; i < size - 1; i++) {
        for (int j = i + 1; j < size; j++) {
            if (str[i] > str[j]) {
                temp = str[i];
                str[i] = str[j];
                str[j] = temp;
            }
        }
    }
}

int manualLength(char *str) {
    int length = 0;
    while (str[length] != '\0') {
        length++;
    }
    return length;
}

int binarySearch(char *str, char item, int min, int max) {
    if (min > max) {
        return -1;
    }
    
    int mid = (min + max) / 2;

    if (str[mid] == item) {
        return mid;
    } 
    else if (str[mid] > item) {
        return binarySearch(str, item, min, mid - 1);
    } 
    else {
        return binarySearch(str, item, mid + 1, max);
    }
}

int main() {
    char str[100];
    char item;
    
    printf("Enter a string: ");
    gets(str);  // Using gets to read the entire string including spaces
    
    printf("Enter a character to search: ");
    scanf(" %c", &item);  // Note the space before %c to skip any newline character
    
    int size = manualLength(str);  // Manually get the length of the string
    sortString(str, size);  // Sort the string
    
    int result = binarySearch(str, item, 0, size - 1);
    
    if (result != -1) {
        printf("The character '%c' is found at index %d.\n", item, result);
    } else {
        printf("The character '%c' is not found in the string.\n", item);
    }
    
    return 0;
}

Question 8: Binary Search with character

#include <stdio.h>
#include <stdlib.h>

void sortString(char *str) {
    char temp;
    for (int i = 0; str[i] != '\0'; i++) {
        for (int j = i + 1; str[j] != '\0'; j++) {
            if (str[i] > str[j]) {
                temp = str[i];
                str[i] = str[j];
                str[j] = temp;
            }
        }
    }
}

int binarySearch(char *str, char item) {
    int min = 0;
    int max = 0;
    while (str[max] != '\0') {
        max++;
    }
    max--;  

    while (min <= max) {
        int mid = (min + max) / 2;
        if (str[mid] == item) {
            return mid;
        } 
        else if (str[mid] < item) {
            min = mid + 1;
        } 
        else {
            max = mid - 1;
        }
    }
    return -1;
}

int main() {
    char *str = (char *)malloc(100 * sizeof(char));
    char item;

    printf("Enter a string: ");
    fgets(str, 100, stdin);

    printf("Enter a character to search: ");
    scanf(" %c", &item);

    sortString(str);

    int result = binarySearch(str, item);

    if (result != -1) {
        printf("The character '%c' is found at index %d.\n", item, result);
    } else {
        printf("The character '%c' is not found in the string.\n", item);
    }

    free(str);
    
    return 0;
}

Question 10: Best and worst complexity of Binary Search

#include <stdio.h>

    int binarySearch(int arr[], int left, int right, int target) {
        while (left <= right) {
            int mid = left + (right - left) / 2;
    
            if (arr[mid] == target) {
                return mid;  // Target found at index mid
            }
            if (arr[mid] < target) {
                left = mid + 1;  // Search in the right half
            } else {
                right = mid - 1; // Search in the left half
            }
        }
        return -1;  // Target not found
    }
    
    int main() {
        int arr[] = {2, 4, 6, 8, 10, 12, 14};
        int size = sizeof(arr) / sizeof(arr[0]);
        int target;
    
        // Best Case Example: Element is the middle element
        target = 8;
        printf("Best Case: Searching for %d\n", target);
        int result = binarySearch(arr, 0, size - 1, target);
        if (result != -1) {
            printf("Element %d found at index %d\n", target, result);
        } else {
            printf("Element %d not found in the array\n", target);
        }
    
        // Worst Case Example: Element is at the end of the array
        target = 14;
        printf("\nWorst Case: Searching for %d\n", target);
        result = binarySearch(arr, 0, size - 1, target);
        if (result != -1) {
            printf("Element %d found at index %d\n", target, result);
        } else {
            printf("Element %d not found in the array\n", target);
        }
    
        // Another Worst Case Example: Element not in the array
        target = 99;
        printf("\nWorst Case: Searching for %d (not in array)\n", target);

            printf("Element %d found at index %d\n", target, result);
        } else {
