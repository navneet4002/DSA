Question 1: Calculate Factorial of a Number Using Recursion

#include<stdio.h>

int factorial(int n)
{
    if(n==0)
    {
        return 1;
    }
    else
    {
        return (n*factorial(n-1));
    }
}

int main()
{
    int n;
    printf("Enter the number for finding factorial :");
    scanf("%d",&n);
    printf("FACTORIAL OF %d = %d",n,factorial(n));
}
OUTPUT

Enter the number for finding factorial :4
FACTORIAL OF 4 = 24

Question 2: Calculate Fibonacci of a Number Using Recursion

#include <stdio.h>

int fibonacci(int n) 
{
    if (n <= 1) 
    {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() 
{
    int n;
    printf("Enter a non-negative integer: ");
    scanf("%d", &n);
    
    if (n < 0) 
    {
        printf("Fibonacci is undefined for negative numbers.\n");
    } 
    else 
    {
        printf("Fibonacci sequence up to %d:\n", n);
        for (int i = 0; i <= n; i++) 
        {
            printf("%d ", fibonacci(i));
        }
        printf("\n");
    }
    
    return 0;
}
OUTPUT

Enter a non-negative integer: 5
Fibonacci sequence up to 5:
0 1 1 2 3 5

Question 3: Calculate the Sum of Natural Numbers Till 10 Using Recursion

#include <stdio.h>

int sumNatural(int n) 
{
    if (n <= 0) 
    {
        return 0;
    }
    return n + sumNatural(n - 1);
}

int main() 
{
    int result = sumNatural(10);
    printf("The sum of natural numbers up to 10 is: %d\n", result);
    return 0;
}
OUTPUT

The sum of natural numbers up to 10 is: 55

Question 4: Convert a Number into Binary Using Recursion

#include <stdio.h>

void printBinary(int n) 
{
    if (n > 1) 
    {
        printBinary(n / 2);

}

Question 5: Find the Largest Element in the Integer Array Using Recursion

#include <stdio.h>

int findLargest(int arr[], int size) 
{
    if (size == 1) 
    {
        return arr[0];
    }
    
    int largest = findLargest(arr, size - 1);
    return arr[size - 1] > largest ? arr[size - 1] : largest;
}

int main() 
{
    int arr[] = {3, 5, 2, 8, 6, 1};
    int size = sizeof(arr) / sizeof(arr[0]);
    
    int largest = findLargest(arr, size);
    printf("The largest element in the array is: %d\n", largest);
    
    return 0;
}
OUTPUT

The largest element in the array is: 8

Question 7: Add All Digits of a Number Entered by the User Using Recursion

#include <stdio.h>

int addDigits(int n) 
{
    if (n == 0) 
    {
        return 0;
    }
    return (n % 10) + addDigits(n / 10);
}

int main() 
{
    int number;
    printf("Enter a number: ");
    scanf("%d", &number);
    
    int sum = addDigits(number);
    printf("The sum of digits of %d is: %d\n", number, sum);
    
    return 0;
}
OUTPUT

Enter a number: 1234
The sum of digits of 1234 is: 10

Question 8: Find the Arithmetic Mean of an Integer Array Using Recursion

#include <stdio.h>

int sumArray(int arr[], int size) 
{
    if (size == 0) 
    {
        return 0;
    }
    return arr[size - 1] + sumArray(arr, size - 1);
}

float findMean(int arr[], int size) 
{
    return (float)sumArray(arr, size) / size;
}

int main() 
{
    int arr[] = {3, 5, 2, 8, 6, 1};
    int size = sizeof(arr) / sizeof(arr[0]);
    
    float mean = findMean(arr, size);
    printf("The arithmetic mean of the array is: %.2f\n", mean);
    
    return 0;
}
OUTPUT

The arithmetic mean of the array is: 4.17
