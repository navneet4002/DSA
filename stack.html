    #include <stdio.h>
    #include <stdlib.h>
        
        void push(char stack[], int size, int *i, char value) 
        {
            if (*i == size - 1) 
            {
                printf("\nThe stack is full!!");
                return;
            } 
            else 
            {
                (*i)++;
                stack[*i] = value;
            }
        }
        
        void pop(char stack[], int size, int *i)
        {
            if (*i == -1) 
            {
                printf("\nThe stack is empty!!");
                return;
            }
            else 
            {
                stack[*i] = '\0';
                (*i)--;
            }
        }
        
        char peek(char stack[], int size, int *i) 
        {
            if (*i == -1) 
            {
                return '\0';  
            }
            else 
            {
                return stack[*i];
            }
        }
        
        int main() 
        {
            char stack[5];
            int i = -1;
            char value;
            int choice;

            {

                printf("\nEnter 4 for exit");
                printf("\nEnter your choice from above options: ");
                scanf("%d", &choice);
        
                if (choice == 1) 
                {
                    printf("\nEnter the Character to be pushed in the stack: ");
                    scanf(" %c", &value);

                    {
                        printf("\nThe character %c is successfully pushed in the Stack at stack[%d]", value, i);

                {
                    if (i == -1) 
                    {
                        printf("\nThe stack is empty!!");
                    } 
                    else 
                    {
                        printf("\nThe character %c is successfully popped from the stack at index %d.", stack[i], i);

                        printf("\nThe peek character in the stack is %c at index %d", top, i);

                else if (choice == 4) 
                {
                    exit(0);
                } 
                else 
                {
                    printf("Invalid choice! Enter another choice...");

Question 3: Stack Operations on Integer Nodes (Dynamically)

    #include<stdio.h>
    #include<stdlib.h>
            
            struct stack{
                int data;
                struct stack *next;
            };
            
            struct stack *start=NULL,*temp,*last;
            
            /// push function to insert the value in stack ///
            
            void push()
            {
                int value;
                if(start==NULL)
                {
                    start=(struct stack*)malloc(sizeof(struct stack));
                    printf("\nEnter the value to be inserted : ");
                    scanf("%d",&value);
                    start->data=value;
                    start->next=NULL;
                    last=start;
                }
                else
                {
                    temp=(struct stack*)malloc(sizeof(struct stack));
                    if(temp!=NULL)
                    {
                        printf("\nEnter the value : ");
                        scanf("%d",&(temp->data));
                        temp->next=last;
                        last=temp;
                    }
                    else
                    {
                        printf("\nMemory is Full!!");
                    }
                
                }
            }
            
            /// pop function to delete the value form stack ///
            
            void pop()
            {
                if(last==NULL)
                {
                    printf("\nStack is empty");
                }
                else
                {
                    printf("\n%d",last->data);
                    temp=last;
                    last=last->next;
                    free(temp);
                }
                if(last==NULL)
                {
                    start=NULL;
                }
            }
            
            /// peek function to print the top value(last value) of the stack ///
            
            void peek()
            {
                if(last==NULL)
                {
                    printf("\nStack is Empty!!");
                }
                else
                {
                    printf("\n%d",last->data);
                }
            }
            
            /// traverse function to print all the values of the stack ///
            
            void traverse()
            {
                if(last==NULL)
                {
                    printf("\nStack is Empty");
                }
                else
                {
                    temp=last;
                    while (temp!= NULL)
                    {
                        printf("\n%d",temp->data);
                        temp=temp->next;
                    }
                }
            }
            
            /// main Function ///
            
            void main()
            {
            
                int choice;
                
                while(1)
                {
                    printf("\nEnter 1 for push()");
                    printf("\nEnter 2 for pop()");
                    printf("\nEnter 3 for peek()");
                    printf("\nEnter 4 for traverse()");
                    printf("\nEnter 5 for exit");
                    printf("\nEnter your choice from above options ");

                    {
                        push();
                        
                           printf("\nThe value is successfully pushed in the Stack");

                    else if(choice==2)
                    {
                        pop();
                        printf("\nThe value is successfully poped from the stack.");
                        
                    }
                    
                    else if(choice==3)
                    {
                        peek();
                    }
                    
                    else if(choice==4)
                    {
                        traverse();
                    }
                    else if(choice==5)
                    {
                        exit(0);

    #include<stdio.h>
    #include<stdlib.h>
            
            struct stack
            {
                char data;
                struct stack *next;
            };
            
            struct stack *start = NULL, *temp, *last;
            
            /// push function to insert the value in stack ///
            void push() 
            {
                char value;
                temp = (struct stack*)malloc(sizeof(struct stack));
                if (temp != NULL)
                {
                    printf("\nEnter the value to be inserted: ");
                    scanf(" %c", &value); 
                    temp->data = value;
                    temp->next = last;
                    last = temp;
            
                    if (start == NULL) 
                    { 
                        start = last;
                    }
                    printf("\nThe value %c is successfully pushed in the Stack", value);
                } 
                else 
                {
                    printf("\nMemory is Full!!");
                }
            }
            
            /// pop function to delete the value from stack ///
            void pop() 
            {
                if (last == NULL) 
                {
                    printf("\nStack is empty");
                } 
                else 
                {
                    printf("\nThe value %c is successfully popped from the stack.", last->data);
                    temp = last;
                    last = last->next;
                    free(temp);
                    
                    if (last == NULL)
                    {
                        start = NULL;
                    }
                }
            }
            
            /// peek function to print the top value (last value) of the stack ///
            void peek()
            {
                if (last == NULL)
                {
                    printf("\nStack is Empty!!");
                } 
                else 
                {
                    printf("\nTop value of the stack: %c", last->data);
                }
            }
            
            /// traverse function to print all the values of the stack ///
            void traverse()
            {
                if (last == NULL) 
                {
                    printf("\nStack is Empty");
                } 
                else 
                {
                    temp = last;
                    printf("\nStack elements are:");
                    while (temp != NULL) 
                    {
                        printf("\n%c", temp->data);
                        temp = temp->next;
                    }
                }
            }

Question 5: Reverse String Using Stacks

    #include<stdio.h>
    #include<string.h>
        
        
        void stringreverse(char stack[],int size,char str[],int len,int *i,int *j)
        {
            for(*j=0;*j<=len-1;(*j)++)
            {
                    (*i)++;
                    stack[*i]=str[*j];
                    //printf("The stack.....");
                    //printf("%c",stack[*i]);
            }
            printf("\nThe string in reverse order : ");
            for(*j=0;*j<=len-1;(*j)++)
            {
                    printf("%c",stack[*i]);
                    stack[*i]==0;
                    (*i)--;
            }
            
        }
            
        
        int stringlen(char str[],int len,int *k)
        {
            for(*k=0;str[*k]!='\0';(*k)++)
            {
                len++;
            }
            return len;
        }
        
        int main()
        {
            char stack[20];
            int i=-1,len=0,j,k;
            char str[20];
            printf("Enter string to reverse : ");
            gets(str);
            puts(str);
            len=stringlen(str,len,&k);
            printf("The string length is %d\n",len);
            stringreverse(stack,20,str,len,&i,&j);
            return 0;
        }

Question 6: Bracket Matching

        #include<stdio.h>
        #include<string.h>
            
        void bracketmatch(char stack[],int size,char str[],int len,int *i,int *j)
            {
                for(*j=0;*j<=len-1;(*j)++)
                {
                    if(str[*j]=='(')
                    {
                        (*i)++;
                        stack[*i]='(';
                        //printf("The stack....");
                        //printf("%c",stack[*i]);
                    }
                    else if(str[*j]==')')
                    {
                        if(*i==-1)
                        {
                            return;
                        }
                        else
                        {
                        stack[*i]==0;
                        (*i)--;
                        }
                    }
                    else
                    {
                        printf("not a bracket!!");
                    }
                }
                
            }
            int stringlen(char str[],int len,int *k)
            {
                for(*k=0;str[*k]!='\0';(*k)++)
                {
                    len++;
                }
                return len;
            }

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct stack 
{
    int max;
    char *array;
    int top;	
};

struct stack s1;
char infix[1000];
char prefix[1000];
int prefix_index = -1;

void init(int size) 
{
    s1.max = size;
    s1.array = (char *) malloc(sizeof(char) * s1.max);
    s1.top = -1;
}

void push(char item) 
{
    if (s1.top == s1.max - 1) 
    {
        printf("\nOverflow!");
        return;
    }
    s1.top++;
    s1.array[s1.top] = item;
}

char peek() 
{
    if (s1.top == -1) 
    {
        printf("\nEmpty Stack, cannot peek!");
        return '\0';
    }
    return s1.array[s1.top];
}

char pop() 
{
    if (s1.top == -1) 
    {
        printf("\nUnderflow!");
        return '\0';
    }
    return s1.array[s1.top--];
}

int precedence(char op) 
{
    if (op == '^') return 3;
    if (op == '*' || op == '/') return 2;
    if (op == '+' || op == '-') return 1;
    return 0;
}

void reverse(char *exp) 
{
    int n = strlen(exp);
    for (int i = 0; i < n / 2; i++) 
    {
        char temp = exp[i];
        exp[i] = exp[n - i - 1];
        exp[n - i - 1] = temp;
    }
}

void convert() 
{
    reverse(infix);
    int i;
    char top_item;
    
    for (i = 0; infix[i] != '\0'; i++) 
    {
        if (isalnum(infix[i])) 
        {
            prefix[++prefix_index] = infix[i];
        } 
        else if (infix[i] == '+' || infix[i] == '-' || infix[i] == '*' || infix[i] == '/' || infix[i] == '^') 
        {
            while (s1.top != -1 && precedence(peek()) > precedence(infix[i])) 
            {
                top_item = pop();
                prefix[++prefix_index] = top_item;
            }
            push(infix[i]);
        } 
        else if (infix[i] == ')') 
        {
            push(infix[i]);
        } 
        else if (infix[i] == '(') 
        {
            while (s1.top != -1 && peek() != ')') 
            {
                prefix[++prefix_index] = pop();
            }
            pop();
        }
    }
    
    while (s1.top != -1) 
    {
        prefix[++prefix_index] = pop();
    }
    
    prefix[prefix_index + 1] = '\0';
    reverse(prefix);
}

{

    gets(infix);

    init(strlen(infix) + 2);
