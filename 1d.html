Question 16: Sort Array

Enter an array from user and sort it in ascending order.
#include <stdio.h>

int main()
{
    int arr[5], i, j, temp;

    for (i = 0; i < 5; i++)
    {
        printf("Enter the element of the array: ");
        scanf("%d", &arr[i]);
    }

    // Sorting the array
    for (i = 0; i < 5; i++)
    {
        for (j = i + 1; j < 5; j++)
        {
            if (arr[i] > arr[j])
            {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    printf("Sorted array in ascending order: ");
    for (i = 0; i < 5; i++)
    {
        printf("%d\t", arr[i]);
    }
    printf("\n");
    return 0;
}

OUTPUT

Enter the element of the array: 4
Enter the element of the array: 2
Enter the element of the array: 5
Enter the element of the array: 1
Enter the element of the array: 3
Sorted array in ascending order: 1	2	3	4	5

Question 20: Find Second Largest

Enter an array from user and find the second largest value.
#include <stdio.h>

int main()
{
    int arr[5];
    int i, largest, second_largest;

    for (i = 0; i < 5; i++)
    {
        printf("Enter the number: ");
        scanf("%d", &arr[i]);
    }

    largest = second_largest = arr[0];
    for (i = 1; i < 5; i++)
    {
        if (arr[i] > largest)
        {
            second_largest = largest;
            largest = arr[i];
        }
        else if (arr[i] > second_largest && arr[i] != largest)
        {
            second_largest = arr[i];
        }
    }

    printf("Second largest value in the array: %d\n", second_largest);
    return 0;
}

OUTPUT

Enter the number: 1
Enter the number: 5
Enter the number: 2
Enter the number: 4
Enter the number: 3
Second largest value in the array: 4
